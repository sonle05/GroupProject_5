# Ứng dụng mô phỏng một Service đơn giản
import os
from flask import Flask, jsonify

app = Flask(__name__)

@app.route('/')
def info():
    # Lấy tên môi trường từ biến hệ thống (Environment Variable)
    # Nếu không có (chạy trên máy thật), mặc định là 'Local Host'
    env_name = os.getenv('ENV_NAME', 'Local Host')
    
    return jsonify({
        "service": "System Integration Lab 1",
        "status": "Healthy",
        "environment": env_name,
        "mechanism": "Containerization (Isolation)"
    })

if __name__ == '__main__':
    # Host='0.0.0.0' để container chấp nhận kết nối từ bên ngoài
    app.run(host='0.0.0.0', port=5000)
